Задание по теме: 0. Видео [#293744]
1. Создайте отдельную базу данных
2. Придумайте и создайте отдельную таблицу. В ней должно быть как минимум 3 атрибута помимо id, причем типы должны быть разными.
3. Добавьте запись, обновите ее и удалите
4. Выполняемые запросы сохраните в отдельном файле

---

Создаем базу данных student

create database student;

---

Создаем таблицу студентов со следуюшими полями:
id, имя, возраст, студент (да, нет)

create table student (
    id serial primary key,
    name varchar(255),
    age smallint,
    isstudent boolean
)

---

Заполняем таблицу студентов

insert into student (name, age, isstudent)
values ('Vasya', 25, true);

----

Проверяем ввод

select * from student;

---

Обновляем данные

update student set age = 28;

---

Опять проверяем что получилось

select * from student;

---

удаляем таблицу

delete from student;
